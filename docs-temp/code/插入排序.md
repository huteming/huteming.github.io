```js
function sort(arr: number[]) {
  for (let i = 1; i < arr.length; i++) {
    let min = i
    for (let j = i - 1; j >= 0; j--) {
      if (arr[j] <= arr[min]) {
        break
      }
      swap(arr, j, min)
      min = j
    }
  }
  return arr
}
function swap(arr: number[], a: number, b: number) {
  const temp = arr[a]
  arr[a] = arr[b]
  arr[b] = temp
}
```
